niveau 0
       regarder_autour
	       1. tourner(2*pi)
	       2. boucle:
	       	  memoriser les items et personnages visibles
		  critere d'arret: tour termine

       deplacement_erratique // par petits deplacements (pas de attendre)
		draw(P(deplacement_t | deplacement_tm1))       
		aller_vers endroit_courant+deplacement

	demander_et_suivre_chemin endroit
	       1. getpath endroit
	       2. boucle: 
		  deplacement_erratique
		  critere d'arret: chemin recu
		
       aller_a endroit // endroit visible ou non
	       1. demander_chemin endroit 
	       2. boucle:
		  suivi_erratique_chemin (chemin)
		  critere d'arret: endroit atteint
		  
	suivi_erratique chemin
		1. aller_vers tete(chemin)
		2. attendre (position atteinte)
		3. suivi_erratique(queue(chemin))

	aller_vers endroit // endroit visible
		1. runto endroit

       maintenir_cible_a_distance // suivi d'un personnage
		draw(P(deplacement | derniere_position(cible)))
	       si cible_visible
	       	       1. draw(P(deplacement | 
			distance_cible vitesse_cible arme_selectionnee
			rotation_relative_cible cible_fait_feu
			ami)) 
				  // deplacement relatif a la cible
		       2. aller_vers endroit_courant+deplacement
	       sinon
			aller_vers(derniere_position_cible)
			attendre(cible visible ou 
				  position atteinte ou 
				  cible morte)
			
       strafe_autour_de_la_cible
	       strafe_tm1 <- strafe_t
	       draw(P(strafe_t | strafe_tm1)) 
				  // strafe relatif a la cible
	       strafe(cible, strafe_t + position_courante)

       attaquer_cible
	       1. se mettre a tirer(cible)
	       2. boucle: 
		  maintenir_cible_a_distance
		  | strafe_autour_de_la_cible
	          critere d'arret: cible_morte ou cible_perdue
	       finalisation: arreter de tirer

       fuire_cible
		1. aller_a item_au_hasard(liste_items)
		   // pas tres satisfaisant, mais bon

       choisir_cible
	       1. chercher_ennemi
	       2. draw(P(ennemi | distance rotation_relative 
				  vitesse_relative arme))

       explorer
	       1. boucle:
		  explorer_local

       explorer_local
	       1. regarder_autour
	       2. aller_vers(plus_proche(noeuds_non_visites)
			ou noeud_au_hasard(noeud_visites))
	       3. attendre_position(noeud)
	       4. memoriser_noeud(noeud)

       chercher_item
	PRE: liste items non vide
	       1. aller_vers(plus_proche(liste_items)) // ou item_au_hasard
	       2. attendre_position(position(item))

       attendre
	       boucle:
	       regarder_autour

       chercher_ennemi
               boucle:
	       explorer_local
	       critere d'arret: ennemi en vue
