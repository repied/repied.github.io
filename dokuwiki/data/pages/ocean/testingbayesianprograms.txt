0 je crée des bindings python pour la classe en question, avec mon Makefile de la mort (le seul truc à modifier est MODULE, qui est le nom de la classe à emballer) (PLDIR doit pointer vers l'installation de PL à utiliser):
 # generic Makefile for swig wrappers of a library
 # Ronan Le Hy, 2003, INRIA

 # customise this only!
 MODULE = heCedric
 
 ifndef MODULE_I
        MODULE_I = $(MODULE).i
 endif

 ifndef LIBMODULE_SO
        LIBMODULE_SO = lib$(MODULE).so
 endif

 ifndef _MODULE_SO
        _MODULE_SO = _$(MODULE).so
 endif

 ifndef MAKE_INTERFACE
        MAKE_INTERFACE = /home/lehy/bin/make-swig-interface
 endif

 ifndef PYTHON
        PYTHON = python2.3
 endif

 ifndef PYTHON_INCLUDE
        PYTHON_INCLUDE = /usr/include/$(PYTHON)
 endif

 ifndef SWIG
        SWIG = /usr/bin/swig
 endif

 ifndef CXX
        CXX = g++-3.2
 endif

 ifndef GEN_DOXYFILE
        GEN_DOXYFILE = /home/lehy/bin/generate-doxyfile
 endif

 ifndef DOXYGEN
        DOXYGEN = doxygen
 endif

 all: python_bindings

 python_bindings: $(_MODULE_SO)
        @echo "Python bindings generated for $(PYTHON)."

 $(MODULE)_doc: $(MODULE).cpp $(MODULE).h
        @echo "Generating documentation for the C++ module in doc/..."
        @mkdir -p doc/
        @$(GEN_DOXYFILE) $(MODULE)
        @$(DOXYGEN) > /dev/null

 $(MODULE).o: $(MODULE).cpp $(MODULE).h 
        @echo "Compiling the generated C++ module..."
        @$(CXX) -c $(MODULE).cpp -I$(PLDIR)/pl/include

 $(LIBMODULE_SO): $(MODULE).o $(MODULE).h $(MODULE)_doc
        @echo "Creating a shared library for your module..."
        @$(CXX) -shared $(MODULE).o -o $(LIBMODULE_SO) -I$(PLDIR)/pl/include -lpl -L$(PLDIR)/lib

 $(MODULE_I): $(MODULE_H)
        @echo "Making an interface file for swig..."
        @$(MAKE_INTERFACE) $(MODULE) > $(MODULE_I)

 $(MODULE)_python_wrap.cpp: $(MODULE_I) $(MODULE).h
        @echo "Running swig to generate Python wrappers..."
        $(SWIG) -module $(MODULE) -c++ -shadow -python -o $(MODULE)_python_wrap.cpp $(MODULE_I) 2> swig.log

 $(MODULE)_python_wrap.o: $(MODULE)_python_wrap.cpp
        @echo "Compiling the Python wrappers..."
        @$(CXX) -I$(PYTHON_INCLUDE) -c $(MODULE)_python_wrap.cpp -o $(MODULE)_python_wrap.o -I$(PLDIR)/pl/include

 $(_MODULE_SO): $(MODULE)_python_wrap.o $(LIBMODULE_SO)
        @echo "Generating a library with your module for Python..."
        @$(CXX) -shared $(MODULE)_python_wrap.o -l$(MODULE) -L$(PLDIR)/lib -L. -o $(_MODULE_SO)

 clean:
        @echo "Cleaning everything..."
        @rm -f $(MODULE).cpp $(MODULE).h $(MODULE).o $(LIBMODULE_SO) $(MODULE_I) $(MODULE)_python_wrap.cpp $(MODULE)_python_wrap_.o $(_MODULE_SO) $(MODULE).pyc $(MODULE).py
        @rm -f warnings_doxygen.log swig.log Doxyfile
        @rm -rf doc/*
